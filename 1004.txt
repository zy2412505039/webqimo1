#include <vector>
using namespace std;

class Solution {
public:
    int longestOnes(vector<int>& nums, int k) {
        int left = 0;
        int zeroCnt = 0;
        int maxLen = 0;
        for (int right = 0; right < nums.size(); ++right) {
            if (nums[right] == 0) {
                zeroCnt++;
            }
            // 窗口0超标，左边界右移
            while (zeroCnt > k) {
                if (nums[left] == 0) {
                    zeroCnt--;
                }
                left++;
            }
            // 更新最大长度
            maxLen = max(maxLen, right - left + 1);
        }
        return maxLen;
    }
};
