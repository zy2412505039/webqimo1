#include <vector>
#include <unordered_map>
using namespace std;

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
private:
    unordered_map<int, int> inPos; // 存储值对应的中序下标
public:
    TreeNode* buildTree(vector<int>& preorder, vector<int>& inorder) {
        int n = inorder.size();
        // 预处理中序值与下标映射
        for (int i = 0; i < n; ++i) {
            inPos[inorder[i]] = i;
        }
        return build(0, n - 1, 0, n - 1, preorder, inorder);
    }

    // 参数说明：
    // pl, pr：前序区间 [pl, pr]
    // il, ir：中序区间 [il, ir]
    TreeNode* build(int pl, int pr, int il, int ir, vector<int>& pre, vector<int>& in) {
        if (pl > pr) return nullptr;
        // 前序第一个为根
        int rootVal = pre[pl];
        TreeNode* root = new TreeNode(rootVal);
        int mid = inPos[rootVal];       // 根在中序的位置
        int leftCnt = mid - il;         // 左子树节点总数
        // 左子树：前序[pl+1, pl+leftCnt]，中序[il, mid-1]
        root->left = build(pl + 1, pl + leftCnt, il, mid - 1, pre, in);
        // 右子树：前序[pl+leftCnt+1, pr]，中序[mid+1, ir]
        root->right = build(pl + leftCnt + 1, pr, mid + 1, ir, pre, in);
        return root;
    }
};
