#include <vector>
#include <unordered_map>
using namespace std;

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
private:
    unordered_map<int, int> inIdx; // 值 -> 中序下标
public:
    TreeNode* buildTree(vector<int>& inorder, vector<int>& postorder) {
        int n = inorder.size();
        // 预处理中序位置映射
        for (int i = 0; i < n; ++i) {
            inIdx[inorder[i]] = i;
        }
        return build(0, n - 1, 0, n - 1, inorder, postorder);
    }

    // il,ir: 中序区间 [il, ir]
    // pl,pr: 后序区间 [pl, pr]
    TreeNode* build(int il, int ir, int pl, int pr, vector<int>& in, vector<int>& post) {
        if (il > ir) return nullptr;
        // 后序末尾是根
        int rootVal = post[pr];
        TreeNode* root = new TreeNode(rootVal);
        int mid = inIdx[rootVal]; // 根在中序的下标
        int leftSize = mid - il;  // 左子树节点数量
        // 左子树：中序[il, mid-1]，后序[pl, pl+leftSize-1]
        root->left = build(il, mid - 1, pl, pl + leftSize - 1, in, post);
        // 右子树：中序[mid+1, ir]，后序[pl+leftSize, pr-1]
        root->right = build(mid + 1, ir, pl + leftSize, pr - 1, in, post);
        return root;
    }
};
