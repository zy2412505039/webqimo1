#include <string>
using namespace std;

class Solution {
public:
    int firstUniqChar(string s) {
        int cnt[26] = {0};
        // 第一次遍历统计次数
        for (char c : s) {
            cnt[c - 'a']++;
        }
        // 第二次遍历找第一个仅出现一次的字符
        for (int i = 0; i < s.size(); i++) {
            if (cnt[s[i] - 'a'] == 1) {
                return i;
            }
        }
        return -1;
    }
};
