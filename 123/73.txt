#include <vector>
using namespace std;

class Solution {
public:
    void setZeroes(vector<vector<int>>& matrix) {
        int m = matrix.size();
        int n = matrix[0].size();
        bool firstRowZero = false, firstColZero = false;

        // 标记首行、首列是否存在0
        for (int j = 0; j < n; ++j) {
            if (matrix[0][j] == 0) {
                firstRowZero = true;
                break;
            }
        }
        for (int i = 0; i < m; ++i) {
            if (matrix[i][0] == 0) {
                firstColZero = true;
                break;
            }
        }

        // 用首行首列标记其余位置的0
        for (int i = 1; i < m; ++i) {
            for (int j = 1; j < n; ++j) {
                if (matrix[i][j] == 0) {
                    matrix[i][0] = 0;
                    matrix[0][j] = 0;
                }
            }
        }

        // 根据标记置零行（从第二行开始）
        for (int i = 1; i < m; ++i) {
            if (matrix[i][0] == 0) {
                for (int j = 1; j < n; ++j)
                    matrix[i][j] = 0;
            }
        }
        // 根据标记置零列（从第二列开始）
        for (int j = 1; j < n; ++j) {
            if (matrix[0][j] == 0) {
                for (int i = 1; i < m; ++i)
                    matrix[i][j] = 0;
            }
        }

        // 处理首行
        if (firstRowZero) {
            for (int j = 0; j < n; ++j)
                matrix[0][j] = 0;
        }
        // 处理首列
        if (firstColZero) {
            for (int i = 0; i < m; ++i)
                matrix[i][0] = 0;
        }
    }
};
