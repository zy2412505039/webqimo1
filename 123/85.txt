#include <vector>
#include <stack>
using namespace std;

class Solution {
private:
    // 复用84.柱状图中最大矩形
    int largestRectangleArea(vector<int>& heights) {
        stack<int> st;
        int maxA = 0;
        heights.push_back(0);
        int n = heights.size();
        for (int i = 0; i < n; ++i) {
            while (!st.empty() && heights[i] < heights[st.top()]) {
                int h = heights[st.top()];
                st.pop();
                int left = st.empty() ? -1 : st.top();
                int w = i - left - 1;
                maxA = max(maxA, h * w);
            }
            st.push(i);
        }
        heights.pop_back(); // 恢复原数组
        return maxA;
    }
public:
    int maximalRectangle(vector<vector<char>>& matrix) {
        if (matrix.empty() || matrix[0].empty()) return 0;
        int col = matrix[0].size();
        vector<int> h(col, 0);
        int res = 0;
        for (auto& row : matrix) {
            // 更新当前行柱状高度
            for (int j = 0; j < col; ++j) {
                if (row[j] == '1') h[j]++;
                else h[j] = 0;
            }
            res = max(res, largestRectangleArea(h));
        }
        return res;
    }
};
