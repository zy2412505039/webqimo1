#include <vector>
#include <unordered_set>
using namespace std;

class Solution {
public:
    bool containsNearbyDuplicate(vector<int>& nums, int k) {
        unordered_set<int> win;
        int n = nums.size();
        for (int i = 0; i < n; ++i) {
            if (win.count(nums[i])) {
                return true;
            }
            win.insert(nums[i]);
            // 窗口超出k个，移除最早的元素
            if (win.size() > k) {
                win.erase(nums[i - k]);
            }
        }
        return false;
    }
};
