#include <vector>
#include <set>
using namespace std;

class Solution {
public:
    bool containsNearbyAlmostDuplicate(vector<int>& nums, int indexDiff, int valueDiff) {
        multiset<long long> st; // long long 防止减法溢出
        int n = nums.size();
        for (int i = 0; i < n; ++i) {
            long long x = nums[i];
            // 找到第一个 >= x - valueDiff 的元素
            auto it = st.lower_bound(x - valueDiff);
            // 该元素存在且不超过 x + valueDiff，则满足条件
            if (it != st.end() && *it <= x + valueDiff) {
                return true;
            }
            st.insert(x);
            // 窗口超过 indexDiff，移除最左侧元素
            if (st.size() > indexDiff) {
                st.erase(st.find(nums[i - indexDiff]));
            }
        }
        return false;
    }
};
