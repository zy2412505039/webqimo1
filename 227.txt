#include <stack>
#include <string>
using namespace std;

class Solution {
public:
    int calculate(string s) {
        stack<int> st;
        int num = 0;
        char op = '+'; // 记录前一个运算符，初始默认第一个数前面是+
        int n = s.size();
        
        for (int i = 0; i < n; ++i) {
            char c = s[i];
            // 拼接多位数
            if (isdigit(c)) {
                num = num * 10 + (c - '0');
            }
            // 到达末尾 或 遇到运算符，执行计算
            if ((!isdigit(c) && c != ' ') || i == n - 1) {
                if (op == '+') {
                    st.push(num);
                } else if (op == '-') {
                    st.push(-num);
                } else if (op == '*') {
                    int top = st.top();
                    st.pop();
                    st.push(top * num);
                } else if (op == '/') {
                    int top = st.top();
                    st.pop();
                    // C++负数除法向0取整，符合题目要求
                    st.push(top / num);
                }
                op = c; // 更新当前运算符
                num = 0; // 重置数字
            }
        }
        // 栈内全部相加
        int res = 0;
        while (!st.empty()) {
            res += st.top();
            st.pop();
        }
        return res;
    }
};
