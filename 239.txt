#include <vector>
#include <deque>
using namespace std;

class Solution {
public:
    vector<int> maxSlidingWindow(vector<int>& nums, int k) {
        deque<int> dq;
        vector<int> res;
        int n = nums.size();
        for (int i = 0; i < n; ++i) {
            // 1. 移除队尾所有 <= nums[i] 的下标，维持单调递减
            while (!dq.empty() && nums[dq.back()] <= nums[i]) {
                dq.pop_back();
            }
            dq.push_back(i);

            // 2. 移除超出窗口左边界的队首
            while (dq.front() < i - k + 1) {
                dq.pop_front();
            }

            // 3. 窗口长度达到k后，记录最大值
            if (i >= k - 1) {
                res.push_back(nums[dq.front()]);
            }
        }
        return res;
    }
};
