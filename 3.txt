#include <string>
#include <unordered_set>
#include <algorithm>
using namespace std;

class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        unordered_set<char> win;
        int left = 0;
        int maxLen = 0;
        for (int right = 0; right < s.size(); ++right) {
            char c = s[right];
            // 存在重复，收缩左边界
            while (win.count(c)) {
                win.erase(s[left]);
                left++;
            }
            win.insert(c);
            maxLen = max(maxLen, right - left + 1);
        }
        return maxLen;
    }
};
