#include <vector>
using namespace std;

class Solution {
public:
    bool isValidSudoku(vector<vector<char>>& board) {
        // 9行、9列、9个九宫格，数字1~9
        bool row[9][9] = {false};
        bool col[9][9] = {false};
        bool box[3][3][9] = {false};

        for (int i = 0; i < 9; ++i) {
            for (int j = 0; j < 9; ++j) {
                char c = board[i][j];
                if (c == '.') continue;
                int num = c - '1'; // 转成0~8下标
                // 冲突判定
                if (row[i][num] || col[j][num] || box[i/3][j/3][num]) {
                    return false;
                }
                // 标记已出现
                row[i][num] = true;
                col[j][num] = true;
                box[i/3][j/3][num] = true;
            }
        }
        return true;
    }
};
