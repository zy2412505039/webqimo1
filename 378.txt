#include <vector>
using namespace std;

class Solution {
public:
    int kthSmallest(vector<vector<int>>& matrix, int k) {
        int n = matrix.size();
        int left = matrix[0][0];
        int right = matrix[n-1][n-1];
        while (left < right) {
            int mid = left + (right - left) / 2;
            // 统计<=mid的元素个数
            int cnt = 0;
            int x = n - 1, y = 0;
            while (x >= 0 && y < n) {
                if (matrix[x][y] <= mid) {
                    cnt += x + 1;
                    y++;
                } else {
                    x--;
                }
            }
            if (cnt < k) {
                left = mid + 1;
            } else {
                right = mid;
            }
        }
        return left;
    }
};
