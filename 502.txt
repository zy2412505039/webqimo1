#include <vector>
#include <stack>
using namespace std;

class Solution {
public:
    vector<int> nextGreaterElements(vector<int>& nums) {
        int n = nums.size();
        vector<int> res(n, -1);
        stack<int> st; // 存下标
        
        // 遍历两倍长度模拟环形
        for (int i = 0; i < 2 * n; ++i) {
            int cur = nums[i % n];
            // 栈顶对应数字 < 当前数字，更新答案
            while (!st.empty() && nums[st.top()] < cur) {
                int idx = st.top();
                st.pop();
                res[idx] = cur;
            }
            // 只压入前n个下标，避免重复处理
            if (i < n) st.push(i);
        }
        return res;
    }
};
