#include <vector>
using namespace std;

class Solution {
private:
    vector<int> parent;
    int find(int x) {
        if (parent[x] != x)
            parent[x] = find(parent[x]);
        return parent[x];
    }
    void unite(int x, int y) {
        x = find(x);
        y = find(y);
        if (x != y)
            parent[y] = x;
    }
public:
    int findCircleNum(vector<vector<int>>& isConnected) {
        int n = isConnected.size();
        parent.resize(n);
        for (int i = 0; i < n; ++i)
            parent[i] = i;
        
        for (int i = 0; i < n; ++i) {
            for (int j = i + 1; j < n; ++j) {
                if (isConnected[i][j] == 1)
                    unite(i, j);
            }
        }
        int cnt = 0;
        for (int i = 0; i < n; ++i)
            if (find(i) == i) cnt++;
        return cnt;
    }
};
