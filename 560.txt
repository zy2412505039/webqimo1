#include <vector>
#include <unordered_map>
using namespace std;

class Solution {
public:
    int subarraySum(vector<int>& nums, int k) {
        unordered_map<long long, int> mp;
        mp[0] = 1; // 前缀和为0，初始出现1次
        long long pre = 0;
        int ans = 0;
        for (int num : nums) {
            pre += num;
            // 累加之前前缀和等于 pre-k 的次数
            ans += mp[pre - k];
            mp[pre]++;
        }
        return ans;
    }
};
