#include <string>
using namespace std;

class Solution {
public:
    int lengthOfLastWord(string s) {
        int len = 0;
        int i = s.size() - 1;
        // 跳过末尾空格
        while (i >= 0 && s[i] == ' ') {
            i--;
        }
        // 统计最后单词字母
        while (i >= 0 && s[i] != ' ') {
            len++;
            i--;
        }
        return len;
    }
};
