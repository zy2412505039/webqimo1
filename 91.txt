#include <string>
#include <vector>
using namespace std;

class Solution {
public:
    int numDecodings(string s) {
        int n = s.size();
        vector<int> dp(n + 1, 0);
        dp[0] = 1; // 空串基准
        
        // 处理第一个字符
        dp[1] = s[0] == '0' ? 0 : 1;
        
        for (int i = 2; i <= n; ++i) {
            // 单独取最后一位
            char c1 = s[i - 1];
            if (c1 != '0') {
                dp[i] += dp[i - 1];
            }
            // 取最后两位
            char c2 = s[i - 2];
            if (c2 == '0') continue; // 前导0两位数非法
            int two = (c2 - '0') * 10 + (c1 - '0');
            if (two <= 26) {
                dp[i] += dp[i - 2];
            }
        }
        return dp[n];
    }
};
