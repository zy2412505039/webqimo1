#include <vector>
#include <string>
#include <unordered_map>
using namespace std;

class Solution {
public:
    string kthDistinct(vector<string>& arr, int k) {
        unordered_map<string, int> cnt;
        // 第一步统计每个字符串出现次数
        for (string& s : arr) {
            cnt[s]++;
        }
        int count = 0;
        // 第二步按原顺序遍历，找第k个只出现一次的串
        for (string& s : arr) {
            if (cnt[s] == 1) {
                count++;
                if (count == k) {
                    return s;
                }
            }
        }
        // 独一无二的字符串不足k个
        return "";
    }
};
